# scripts/generate_badges.py
import cairosvg
import os

badges = {
    "family": "assets/badges/family-badge.svg",
    "tianshu": "assets/badges/tianshu-badge.svg",
    "shouhu": "assets/badges/shouhu-badge.svg",
    # ... 所有成员
}

output_dir = "public/badges"
os.makedirs(output_dir, exist_ok=True)

for name, svg_path in badges.items():
    png_path = f"{output_dir}/{name}.png"
    cairosvg.svg2png(url=svg_path, write_to=png_path, output_width=800, output_height=160)
    print(f"🌹 生成 {png_path}")14.3 集成至 CI（GitHub Actions）

若项目文档需生成 PDF（如 Docusaurus、mdpdf），可使用 pdftk 或 qpdf 叠加水印页。
示例：使用 Python + reportlab 生成水印 PDF 并叠加

# scripts/pdf_watermark.py

from reportlab.pdfgen import canvas
from reportlab.lib.pagesizes import A4
import subprocess, sys

watermark_pdf = "watermark.pdf"
c = canvas.Canvas(watermark_pdf, pagesize=A4)

📱 十九、移动端定制水印逻辑
移动端特性：屏幕窄小、触摸交互、性能敏感、安全区域差异。水印需自适应旋转、字号缩放、不影响触摸、不遮挡内容。
19.1 核心思路
• 使用 canvas 生成水印图片，转为 base64 作为 background-image，或通过 CSS ::after 伪元素叠加。
• 监听 resize 和 orientationchange 重新绘制，保证横竖屏清晰。
• 文字大小基于视口宽度 vw 单位动态计算，旋转角度为 -25deg，透明度 0.06。
• 对 pointer-events: none 保证交互穿透。
19.2 移动端水印组件（React/TSX 示例）
// components/MobileWatermark.tsx
import { useEffect, useRef } from 'react';

const MOTTO = '人从众曌众从人 · YYC³ AI Family 🌹';

export default function MobileWatermark() {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  const draw = () => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    const dpr = window.devicePixelRatio || 1;
    const w = window.innerWidth;
    const h = window.innerHeight;
    canvas.width = w * dpr;
    canvas.height = h * dpr;
    canvas.style.width = `${w}px`;
    canvas.style.height = `${h}px`;
    ctx.scale(dpr, dpr);

    ctx.fillStyle = 'rgba(94, 44, 138, 0.06)';
    ctx.font = `bold ${w * 0.045}px "PingFang SC", "Microsoft YaHei", sans-serif`;
    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';

    const stepX = w * 0.45;
    const stepY = h * 0.35;
    for (let y = -stepY; y < h + stepY; y += stepY) {
      for (let x = -stepX; x < w + stepX; x += stepX) {
        ctx.save();
        ctx.translate(x, y);
        ctx.rotate((-20 * Math.PI) / 180);
        ctx.fillText(MOTTO, 0, 0);
        ctx.restore();
      }
    }
  };

  useEffect(() => {
    draw();
    const handle = () => draw();
    window.addEventListener('resize', handle);
    window.addEventListener('orientationchange', handle);
    return () => {
      window.removeEventListener('resize', handle);
      window.removeEventListener('orientationchange', handle);
    };
  }, []);

return (
    <canvas
      ref={canvasRef}
      style={{
        position: 'fixed',
        top: 0,
        left: 0,
        width: '100vw',
        height: '100dvh', // 动态视口高度，避开工具栏
        pointerEvents: 'none',
        zIndex: 9999,
      }}
    />
  );
}2.3 纯 CSS 降级方案（若不引入 Canvas）
